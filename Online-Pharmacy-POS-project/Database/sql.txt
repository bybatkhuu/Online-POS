name, barcode, serial_number, price

SELECT "item_ids"."name", "item_ids"."barcode", "serials"."serial_number", "item_ids"."item_id" FROM
	(SELECT "items"."name", "barcodes"."barcode", "barcodes"."item_id" FROM (SELECT "id", "name" FROM pos_item_names WHERE "name" LIKE concat('А', '%')) AS items
	INNER JOIN
	(SELECT "item_id", "barcode" FROM pos_barcodes WHERE "barcode" LIKE concat('0','%')) AS barcodes
	ON "items"."id" = "barcodes"."item_id") AS item_ids
INNER JOIN
	(SELECT "pos_item_serials"."serial_number", "branches"."price", "branches"."item_id" FROM pos_item_serials
	INNER JOIN
	(SELECT "serial_id", "price", "item_id" FROM pos_branch_prices WHERE "branch_id" = 1) AS branches
	ON "pos_item_serials"."item_id" = "branches"."item_id") AS serials
ON "item_ids"."item_id" = "serials"."item_id" ORDER BY "item_ids"."item_id";
